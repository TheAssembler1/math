\begin{tcolorbox}[title=Problem 2, breakable]
    Prove that every subgroup of a cyclic group is cyclic.
\end{tcolorbox}

\begin{proof}
    Let $G$ be a cyclic group.
    Suppose $H$ is a subgroup of $G$.
    If $H$ only contains the identity then clearly it is cyclic.
    Thus, suppose $H$ has more than one element.
    Let $\langle g \rangle = G$ and $g^m = x \in H$ where $m$ is the smallest positive integer.
    Furthermore, let $h$ be an arbitrary element in $H$.
    Then $h = g^k$ for some $k \in \mathbb{Z}$.
    We can apply the Division Algorithm to $k$ and $m$ to write
    \[k = q m + r \text{ for some integer } 0 \le r < m.\]
    Then 
    \[h = g^k = g^{q m + r} = (g^m)^q \cdot g^r = x^q \cdot g^r.\]
    Since $h \in H$ and $x^q \in H$, we have $g^r \in H$.
    By minimality of $m$, $r = 0$, so $h = x^q \in \langle x \rangle$.
\end{proof}

\begin{tcolorbox}[title=Problem 4, breakable]
    Suppose that $g$ is an element of an infinite order 
    in a group $G$. Prove that no two distinct powers of $g$
    (with any integer exponent) are equal.
\end{tcolorbox}

\begin{proof}
    For contradiction, suppose there exists $q, r$ such that $q \ne r$ and $g^q = g^r$.
    Furthermore, suppose wlog $r < q$. Then 
    \[g^q = g^r \iff g^q \cdot g^{-r} = 1 \iff g^{q - r} = 1 \iff q - r = 0 \iff q = r,\]
    which is a contradiction.
\end{proof}


\begin{tcolorbox}[title=Problem 5, breakable]
    If $a$ and $b$ are elements of a group that commute and $\langle a \rangle \cap \langle b \rangle = \{1\}$,
    what is the order of $ab$ if the order of $a$ is $m$ and the order of $b$ is $n$?
    Prove your assertion. Show by example that your assertion is false in general,
    in the case that $a$ and $b$ do not commute.
\end{tcolorbox}

\textbf{Solution: } The order of $ab$ is $mn$.

\begin{proof}
    Suppose $a$ and $b$ are elements of a group that commute and $\langle a \rangle \cap \langle b \rangle = \{1\}$.  
    Since $a$ and $b$ commute, we have
    \[
    (ab)^{mn} = a^{mn} b^{mn} = (a^m)^n (b^n)^m = 1^n \cdot 1^m = 1.
    \]
    Now suppose there exists $k < mn$ such that $(ab)^k = 1$.  
    Then, using commutativity,
    \[
    1 = (ab)^k = a^k b^k \implies a^k = (b^k)^{-1} \in \langle b \rangle.
    \]  
    But $a^k \in \langle a \rangle$, so $a^k \in \langle a \rangle \cap \langle b \rangle = \{1\}$.  
    Thus $a^k = 1$ and $b^k = 1$.  
    Now, $m \mid k$ and $n \mid k$.  
    The smallest positive $k$ divisible by both $m$ and $n$ is $mn$.  
    Therefore no $k < mn$ satisfies $(ab)^k = 1$, so the order of $ab$ is $mn$.
\end{proof}

\begin{tcolorbox}[title=Problem 6, breakable]
    If $a$ and $b$ are elements of a group whose orders are relatively prime,
    what can you say about $\langle a \rangle \cap \langle b \rangle$? Prove your assertion.
\end{tcolorbox}

\begin{proof}
    Suppose $a$ and $b$ are elements of a group whose orders are relatively prime.
    Let $o(a) = m$ and $o(b) = n$ such that $gcd(m, n) = 1$.
    Now suppose $x \in \langle a \rangle \cap \langle b \rangle$.
    Then $x = a^k \in \langle a \rangle$ and $x = b^{k'} \in \langle b \rangle$.
    Thus, by Theorem 26.2, $o(x) \mid m$ and $o(x) \mid n$.
    But since $gcd(m, n) = 1$, we have $o(x) = 1$.
    Thus $x$ is the identity element.
    So $\langle a \rangle \cap \langle b \rangle = \{1\}$.
\end{proof}

\begin{tcolorbox}[title=Problem 7, breakable]
    Prove that if $G$ is a finite cyclic group with more than two elements, then $G$ has more than one generator.
\end{tcolorbox}

\begin{proof}
    Suppose $G$ is a finite cyclic group with more than two elements and let $G=\langle g\rangle$.
    Let $o(g)=n>2$. Then $g^{-1}=g^{n-1}$.
    Since $n>2$, we have $g^{-1}\neq g$.
    Now let $x \in G$, so $x = g^k$ for some integer $k$.  
    Then, since $g^n = e$ we have $x = (g^{-1})^{\,n-k}$.
    Thus every element of $G$ can be written as a power of $g^{-1}$.
    Therefore, $\langle g^{-1} \rangle = G$, and $G$ has more than one generator.
\end{proof}

\begin{tcolorbox}[title=Problem 10, breakable]
    Show that if $G$ is a cyclic group of order $m$ and $n$ divides $m$, then $G$
    has a subgroup of order $n$. (This subgroup will itself be cyclic.)
\end{tcolorbox}

\begin{proof}
    Suppose $G$ is a cyclic group of order $m$ and $n$ divides $m$.
    Now 
    \[G = \langle g \rangle = \{ g^0, g^1, \ldots, g^{m-1} \}.\]
    Since $m = n k$ for some $k \in \mathbb{Z}$, consider 
    \[G' = \{g^{m/n \cdot i} \mid i = 0, 1, \dots, n-1\} = \{ g^0, g^k, \ldots, g^{(n - 1)k} \}.\]
    Clearly, with $i = 0$, we have $g^0 = e \in G' \ne \emptyset$.
    Now, for any $g^{m/n \cdot i} \in G'$, its inverse is $g^{-m/n \cdot i} = g^{m - m/n \cdot i} \in G'$. 
    Finally, let $f, r \in G'$. Then $f = g^{m/n \cdot i}$ and $r = g^{m/n \cdot j}$ for some $i, j$.
    So 
    \[f \cdot r = g^{m/n \cdot (i+j)} = g^{m/n \cdot ((i+j) \bmod n)} \in G'.\]
    Therefore, $G'$ is a subgroup of $G$.
\end{proof}

\begin{tcolorbox}[title=Problem 11, breakable]
    Generalizing the group given in Exerciwse 24.7, let $\rho_{i, n}$ be the 
    rotation of the circle counterclockwise through an angle of $2 \pi i / n$,
    for $i = 1, 2, \ldots, n - 1$ and all $n \ge 2$. Let $e$ be the `identity'
    rotation. Show that this set of rotations forms a group under composition.
    Show that all elements in this group are torsion (even though the group is infinite).
    Show that each finite subgroup is cyclic.
\end{tcolorbox}

\begin{proof}
    Let $\rho_{i, n}$ be the rotation of the circle counterclockwise through an angle $2 \pi i / n$, for $i = 0,1, \dots, n-1$, and let $e = \rho_{0,n}$ be the identity rotation. Consider any 
    \(\rho = \rho_{k,n}, \ \mu = \rho_{k',n'}, \ \psi = \rho_{k'',n''}\).  
    Clearly, composition is associative
    \[
        \rho \circ (\mu \circ \psi) = (\rho \circ \mu) \circ \psi.
    \]
    For closure, note that
    \[
        \rho_{k,n} \circ \rho_{k',n'} = \rho_{i/N, N}, \quad N = \mathrm{lcm}(n,n'), \ i = k \frac{N}{n} + k' \frac{N}{n'},
    \]
    so the composition is still a rotation by a rational multiple of $2\pi$.  
    For inverses, each $\rho_{k,n}$ has
    \[
        \rho_{k,n}^{-1} = \rho_{n-k,n},
    \]
    which is also a rational rotation.
    Each element is torsion, since for any $\rho_{k,n}$,
    \[
        (\rho_{k,n})^n = \rho_0 = e.
    \]
    Finally, any finite subgroup consists of rotations with a common denominator. 
    Choosing the rotation with the smallest positive angle in the subgroup, all other elements are powers of this rotation. Thus, each finite subgroup is cyclic.
\end{proof}
\begin{tcolorbox}[title=Problem 13, breakable]
    If $G$ is a finite group where every non-identity element is a generator of $G$,
    what can you say about the order of $G$? Prove your assertion.
\end{tcolorbox}

\begin{theorem}
    If $G$ is a finite group where every non-identity element is a generator of $G$,
        then $|G|$ is prime.
\end{theorem}

\begin{proof}
    Suppose $G$ is a finite group where every non-identity element is a generator of $G$.
    Let $\langle g \rangle = G$ so 
    \[
        G = \langle g \rangle = \{g^0, g^1, \ldots, g^{n - 1}\}.
    \]
    For contradiction, suppose $n$ is not prime, so there exists a divisor $p$ with $1 < p < n$.
    Write $n = p k$ for some integer $k > 1$.
    Then 
    \[
        o(g^p) = \frac{n}{p} = k < n,
    \]
    so $g^p$ cannot generate $G$, which is a contradiction.
    Therefore, there is no divisor $p$ with $1 < p < n$, and thus $n$ is prime.
\end{proof}
