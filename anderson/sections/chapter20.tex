\begin{tcolorbox}[title=Problem 2, breakable]
    Prove Theorem 20.3. That is, suppose the $R$ is a commutative ring,
    and $0 \ne a \in R$. Prove that $\langle a \rangle$
    is a prime ideal if and only if $a$ is a prime element.
\end{tcolorbox} 

\begin{proof}
    Suppose $\langle a \rangle$ is a prime ideal.
    Now, suppose $a \mid xy$.
    Then $at = xy$ for some $t \in R$, so $xy \in \langle a \rangle$.
    We must show either $a \mid x$ or $a \mid y$.
    Since $\langle a \rangle$ is a prime ideal, we have $x \in \langle a \rangle$ or $y \in \langle a \rangle$.
    Suppose, WLOG, $x \in \langle a \rangle$.
    Then $x = ak$ for some $k \in R$, so $a \mid x$, as required.

    Conversely, suppose $a$ is a prime element.
    Let $xy \in \langle a \rangle$.
    Then $xy = ak$ for some $k \in R$.
    We must show either $x \in \langle a \rangle$ or $y \in \langle a \rangle$.
    So $a \mid xy$, and, since $a$ is prime, $a \mid x$ or $a \mid y$.
    Suppose, WLOG, $a \mid x$.
    So $x = af$ for some $f \in R$.
    It follows that $x \in \langle a \rangle$,
        so $\langle a \rangle$ is a prime ideal, as required.
\end{proof}

\begin{tcolorbox}[title=Problem 4, breakable]
    Find all prime ideals of $\mathbb{Z}$. Find all maximal ideals of $\mathbb{Z}$.
\end{tcolorbox} 

\textbf{Solution: } The ideals of the form $\langle p \rangle$
(all ideals are of this form since $\mathbb{Z}$ is a PID) where $p$ is prime (which corresponds to irreducibility in $\mathbb{Z}$)
are maximal, and these are all of the maximal ideals.
The prime ideals are exactly the maximal ideals together with $\langle 0 \rangle$.

\begin{tcolorbox}[title=Problem 12, breakable]
    Consider the function 
    \[\rho : \mathbb{Z}[x] \longrightarrow \mathbb{Z}_3 [x]\]
    defined by 
    \[\rho(a_0 + a_1 x + \cdots + a_n x^n) = [a_0] + [a_1]x + \cdots + [a_n]x^n.\]
    \begin{enumerate}
        \item Prove that this is an onto ring homomorphism.
        \item Why is the kernel of $\rho$ a prime ideal? Give an explicit description of its kernel (it is a principal ideal).
    \end{enumerate}
\end{tcolorbox} 

\begin{proof}
    Let $f = \sum a_i x^i$ and
    $g = \sum b_i x^i$ in $\mathbb{Z}[x]$, the $i$th coefficient of $f + g$
    is $a_i + b_i$, and $[a_i + b_i] = [a_i] + [b_i]$, so
    $\rho(f + g) = \rho(f) + \rho(g)$.
    Similarly, the $k$th coefficient of $fg$ is $\sum_{i+j=k} a_i b_j$, and
    $\left[\sum_{i+j=k} a_i b_j\right] = \sum_{i+j=k} [a_i][b_j]$,
    so $\rho(fg) = \rho(f)\rho(g)$.
    Finally, $\rho(1) = [1]$.

    Now, let $[b_0] + [b_1]x + \cdots + [b_n]x^n \in \mathbb{Z}_3[x]$,
    where $b_i \in \mathbb{Z}$.
    Notice, $b_0 + b_1 x + \cdots + b_n x^n \in \mathbb{Z}[x]$ and
    $\rho(b_0 + b_1 x + \cdots + b_n x^n) = [b_0] + [b_1]x + \cdots + [b_n]x^n$.
    Thus $\rho$ is onto.

    Now, suppose $\rho(a_0 + a_1 x + \cdots + a_n x^n) = [a_0] + [a_1]x + \cdots + [a_n]x^n = 0$.
    It follows that $a_i \equiv 0 \pmod{3}$ for each $i$, so every coefficient
    is divisible by $3$. Conversely, any polynomial with all coefficients
    divisible by $3$ maps to $0$.
    Thus $\ker \rho = \langle 3 \rangle$, the principal ideal generated by $3$
    in $\mathbb{Z}[x]$.
    Since $\rho$ is onto, the First Isomorphism Theorem gives
    $\mathbb{Z}[x]/\ker \rho \cong \mathbb{Z}_3[x]$, which is an integral
    domain because $\mathbb{Z}_3$ is. Therefore $\ker \rho$ is a prime ideal.
\end{proof}
