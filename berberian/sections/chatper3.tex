\subsection{Linear Subspace Generated by a Subset}

\begin{tcolorbox}[title=Problem 1, breakable]
    If $T : V \longrightarrow W$ is linear and $A$
    is a subset of $V$ such that $T(A)$ is generating 
    for $W$, then $T$ is surjective.
\end{tcolorbox}

\begin{proof}
    Let $w$ be an arbitrary element in $W$.
    Since $T(A)$ generates $W$, $w$ is a linear combination of elements in $T(A)$.
    Suppose 
    \[
    w = c_1 a_1 + \cdots + c_n a_n \text{ such that } a_1, \ldots, a_n \in T(A).
    \]
    But $a_i = T(v_i)$ for some $v_i \in V$, so 
    \[
    c_1 a_1 + \cdots + c_n a_n = c_1 T(v_1) + \cdots + c_n T(v_n) = T(c_1 v_1) + \cdots + T(c_n v_n) = T(c_1 v_1 + \cdots + c_n v_n).
    \]
    The last two equalities hold since $T$ is linear.
    Thus $T$ is surjective.
\end{proof}

\begin{tcolorbox}[title=Problem 2, breakable]
    If $M$ and $N$ are linear subspaces of $V$ such that $V = M + N$ ($1.6.9$)
    and $T : V \longrightarrow V / N$ is the quotient mapping, then 
    $T(M) = V / N$ but $M$ need not equal $V$; thus we cannot conclude in 
    Exercise 1 that $A$ is a generating for $V$. Can you exhibit an example
    of such $V, M$ and $N$?
\end{tcolorbox}

\begin{proof}
    Consider $V = \mathbb{R}$, $M = \{0\}$, and $N = \mathbb{R}$. Then $V = M + N$, but $M \ne V$.
    Let $T : V \to V/N$ be the quotient map. Since $V/N = \{0\}$, we have $T(M) = \{0\} = V/N$.
\end{proof}

\begin{tcolorbox}[title=Problem 3, breakable]
    (Substitution Principle) Suppose $x, x_1, \ldots, x_n$ are 
    vectors such that $x$ is a linear combination of $x_1, \ldots, x_n$
    but not of $x_1, \ldots, x_{n - 1}$. Prove:
    \[[\{x_1, \ldots, x_n\}] = [\{x_1, \ldots, x_{n - 1}, x\}].\]
\end{tcolorbox}

\begin{proof}
    Since $x$ is a linear combination of $x_1, \ldots, x_n$ suppose 
    \[
    x = c_1 x_1 +\cdots + c_n x_n.
    \]
    Since $x$ is not a linear combination of $x_1, \ldots, x_{n-1}$, we have $c_n \neq 0$, so
    \[
    x_n = \frac{1}{c_n}\big(x - c_1 x_1 - \cdots - c_{n-1} x_{n-1}\big).
    \]
    Let $v \in [\{x_1, \ldots, x_n\}]$ and suppose 
    \[
    v = d_1 x_1 + \cdots + d_n x_n.
    \]
    Substituting for $x_n$, we get
    \[
    v = d_1 x_1 + \cdots + d_{n-1} x_{n-1} + d_n \frac{1}{c_n}\big(x - c_1 x_1 - \cdots - c_{n-1} x_{n-1}\big)
    \in [\{x_1, \ldots, x_{n-1}, x\}].
    \]
    Conversely, let $v \in  [\{x_1, \ldots, x_{n-1}, x\}]$ and suppose 
    \[
    v = d_1 x_1 + \cdots + d_{n - 1} x_{n - 1} + d x.
    \]
    Then 
    \[
    x = c_1 x_1 +\cdots + c_n x_n \implies dx = (d c_1) x_1 + \cdots + (d c_n) x_n.
    \]
    Thus 
    \[
    v = d_1 x_1 + \cdots + d_{n - 1} x_{n - 1} + (d c_1) x_1 + \cdots + (d c_n) x_n \in [\{x_1, \ldots, x_n\}].
    \]
\end{proof}

\begin{tcolorbox}[title=Problem 4, breakable]
    Let $S : V \longrightarrow W$ and $T : V \longrightarrow W$
    be linear mappings, and let $A$ be a subset of $V$ such that $V = [A]$.
    Prove that if $Sx = Tx$ for all $x \in A$, then $S = T$. (Informally,
    a lienar mapping is determined by its values on a generating set.)
\end{tcolorbox}

\begin{proof}
    Suppose $Sx = Tx$ for all $x \in A$.
    Let $x' \in V$ and since $V = [A]$ so 
    \[
    x' = c_1 a_1 + \cdots + c_n a_n.
    \]
    Then 
    \begin{align*}
        T(x') &= T(c_1 a_1 + \cdots + c_n a_n) \\
        &= T(c_1 a_1) + \cdots + T(c_n a_n) \\
        &= c_1 T(a_1) + \cdots + c_n T(a_n) \\
        &= c_1 S(a_1) + \cdots + c_n S(a_n) \\
        &= S(c_1 a_1) + \cdots + S(c_n a_n) \\
        &= S(c_1 a_1 + \cdots + c_n a_n) \\
        &= S(x').
    \end{align*}
    Thus $Sx' = Tx'$ for all $x' \in V$, so $S = T$.
\end{proof}

\begin{tcolorbox}[title=Problem 5, breakable]
    Let $V$ be a vector space, $A$ a nonempty subset of $V$. For each finite
    subset $\{x_1, \ldots, x_n\}$ of $A$, let 
    \[M(x_1, \ldots, x_n) = \{a_1 x_1 + \cdots + a_n x_n \mid a_1, \ldots, a_n \text{ scalars.}\};\]
    as noted in the proof of Corollary $3.1.5$, this is a linear subspace of $V$.
    Let $\mathcal{N}$ be the set of all linear subspaces of $V$ obtained in this way
    (by considering all possible finite subsets of $A$); by definition,
    $[A] = \bigcup \mathcal{N}$ (the union of the subspaces belonging to $\mathcal{N}$).
    Prove that $[A]$ is a linear subspace of $V$ by verifying criterion 1.6, Exercise 9.
\end{tcolorbox}

\begin{proof}
    We must show that if $M, K \in \mathcal{N}$ then there exists $P \in \mathcal{N}$
    such that $M \subset P$ and $K \subset P$.
    Therefore suppose $M, K \in \mathcal{N}$. Then there exist finite subsets
    $\{x_1, \ldots, x_n\} \subseteq A$ and $\{y_1, \ldots, y_m\} \subseteq A$
    such that
    \[
        M = M(x_1, \ldots, x_n), \quad K = M(y_1, \ldots, y_m).
    \]
    Let
    \[
        P = M(x_1, \ldots, x_n, y_1, \ldots, y_m).
    \]
    Then $P \in \mathcal{N}$, and clearly $M \subset P$ and $K \subset P$.
    From 1.6 Exercise 9 it follows that $\bigcup \mathcal{N}$ is a linear subspace of $V$.
\end{proof}

\begin{tcolorbox}[title=Problem 6, breakable]
    Let $V$ and $W$ be vector spaces, $V \times W$ the product vector space ($1.3.11$);
    let $A \subset V$ and $B \subset W$. Prove:
    \begin{enumerate}
        \item $[A \times B] \subset [A] \times [B]$,
        \item In general, $[A \times B] \ne [A] \times [B]$, {Hint: Consider $A = \emptyset, B = W$; or $V = W = R$ and $A = B = \{1\}$.}
        \item $[(A \times \{\emptyset\}) \cup (\{\emptyset\} \times B)] = [A] \times [B]$. {Hint 
        \[\left(\sum_{i = 1}^{m} c_i (x_i, 0) + \sum_{j = 1}^{n} d_j (0, y_j) \right).\]    
        }
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    Let $(x, y)$ be an arbitrary element in $[A \times B]$.
    Then
    \[
    (x, y) = c_1 (a_1, b_1) + \cdots + c_n (a_n, b_n)
    \]
    for some $(a_i, b_i) \in A \times B$.
    But then 
    \[
    (x, y) = (c_1 a_1 + \cdots + c_n a_n,\; c_1 b_1 + \cdots + c_n b_n).
    \]
    Thus
    \[
    x = c_1 a_1 + \cdots + c_n a_n \in [A], \quad
    y = c_1 b_1 + \cdots + c_n b_n \in [B],
    \]
    so $(x, y) \in [A] \times [B]$.
\end{proof}

\begin{proof}
    Let $V = W = \mathbb{R}$ and $A = B = \{1\}$.
    Then
    \[
    A \times B = \{(1,1)\},
    \]
    so
    \[
    [A \times B] = \{c(1,1) \mid c \in \mathbb{R}\}
    = \{(c,c) \mid c \in \mathbb{R}\}.
    \]
    Conversely
    \[
    [A] = \mathbb{R}, \quad [B] = \mathbb{R},
    \]
    so
    \[
    [A] \times [B] = \mathbb{R} \times \mathbb{R}.
    \]
    Since
    \[
    \{(c,c) \mid c \in \mathbb{R}\} \ne \mathbb{R} \times \mathbb{R},
    \]
    it follows that
    \[
    [A \times B] \ne [A] \times [B].
    \]
\end{proof}

\begin{proof}
    Let $(x,y) \in [(A \times \{0\}) \cup (\{0\} \times B)]$.
    Then $(x,y)$ is a linear combination of elements of the form $(a,0)$ and $(0,b)$ so
    \[
    (x,y) = \sum_{i=1}^m c_i (x_i,0) + \sum_{j=1}^n d_j (0,y_j),
    \]
    where $x_i \in A$ and $y_j \in B$.
    Then
    \[
    (x,y) = \left(\sum_{i=1}^m c_i x_i,\; \sum_{j=1}^n d_j y_j \right),
    \]
    so $x \in [A]$ and $y \in [B]$. Thus
    \[
    [(A \times \{0\}) \cup (\{0\} \times B)] \subset [A] \times [B].
    \]
    Conversely let $(x,y) \in [A] \times [B]$. Then
    \[
    x = \sum_{i=1}^m c_i x_i, \quad y = \sum_{j=1}^n d_j y_j,
    \]
    with $x_i \in A$, $y_j \in B$. Then
    \[
    (x,y) = \sum_{i=1}^m c_i (x_i,0) + \sum_{j=1}^n d_j (0,y_j),
    \]
    which shows $(x,y) \in [(A \times \{0\}) \cup (\{0\} \times B)]$.
    Thus
    \[
    [(A \times \{0\}) \cup (\{0\} \times B)] = [A] \times [B].
    \]
\end{proof}

\begin{tcolorbox}[title=Problem 1, breakable]
    Let $V = \mathcal{F}(\mathbb{R})$ be the vector 
    space of all real-valued functions of a real 
    variable (Example 1.3.4). Define $f : \mathbb{R} \longrightarrow \mathbb{R}$
    by $f(t) = \sin(t + \pi / 6)$. Prove that the function $f$,
    $\sin$, $\cos$ are linearly dependent.
\end{tcolorbox}

\begin{proof}
    Simply note 
    \begin{align*}
        f(t) &= \sin(t + \pi/6) \\
             &= \sin(t) \cos(\pi/6) + \cos(t)\sin(\pi/6) \\
             &= \frac{\sqrt{3}}{2} \sin(t) + \frac{1}{2} \cos(t).
    \end{align*}
\end{proof}

\begin{tcolorbox}[title=Problem 2, breakable]
    In the vector space of real polynomial 
    function (Example 1.3.5), show that the 
    function $p_1, p_2, p_3, p_4$ defined by 
    $p_1(t) = t^2 + t + 1$,
    $p_2(t) = t^2 + 5$,
    $p_3(t) = 2t + 1$,
    $p_4(t) = 3t^2 + 5t$ are linearly dependent (cf. 1.5.3).
\end{tcolorbox}

\begin{proof}
    Simply note
    \begin{align*}
        -35p_1(t) + 8p_2(t) - 5p_3(t) + 9p_4(t)
        &= -35(t^2+t+1) + 8(t^2+5) - 5(2t+1) + 9(3t^2+5t) \\
        &= (-35t^2 -35t -35) + (8t^2 + 40) + (-10t -5) + (27t^2 + 45t) \\
        &= 0.
    \end{align*}
\end{proof}

\begin{tcolorbox}[title=Problem 5, breakable]
    If $T : V \longrightarrow W$ is a linear mapping 
    and $x_1, \ldots, x_n$ are linearly dependent 
    vectors in $V$, then the vectors $Tx_1, \ldots, Tx_n$
    of $W$ are linearly dependent.
\end{tcolorbox}

\begin{proof}
    Since $x_1,\ldots,x_n$ are linearly dependent, there exist scalars
    $c_1,\ldots,c_n$, not all zero, such that
    \[c_1 x_1 + \cdots + c_n x_n = 0 \implies T(c_1 x_1 + \cdots + c_n x_n) = T(0) = c_1 T(x_1) + \cdots + c_n T(x_n) = 0.\]
\end{proof}

\begin{tcolorbox}[title=Problem 6, breakable]
    A finite list of vectors $x_1, \ldots, x_n$ 
    is linearly dependent if and only if 
    there exists an index $j$ such that 
    $x_j \in [\{x_i \mid i \ne j\}]$ (cf. 3.1.5).
\end{tcolorbox}

\begin{proof}
    \underline{\textbf{($\longrightarrow$)}}
    Suppose a finite list of vectors $x_1, \ldots, x_n$ 
    is linearly dependent. We have $c_1, \ldots, c_n$ not all zero such that
    \[c_1 x_1 + \cdots + c_n x_n = 0.\]
    Wlog suppose $c_j \ne 0$ for some $1 \le j \le n$.
    Then
    \[
    x_j = \frac{-1}{c_j}(c_1 x_1 + \cdots + c_{j-1} x_{j-1} + c_{j+1} x_{j+1} + \cdots + c_n x_n)
    \in [\{x_i \mid i \ne j\}].
    \]

    \underline{\textbf{($\longleftarrow$)}}
    Suppose there exists an index $j$ such that 
    $x_j \in [\{x_i \mid i \ne j\}]$.
    Then we have 
    \[x_j = c_1 x_1 + \cdots + c_{j-1} x_{j-1} + c_{j+1} x_{j+1} + \cdots + c_n x_n.\]
    Thus
    \[-x_j + c_1 x_1 + \cdots + c_{j-1} x_{j-1} + c_{j+1} x_{j+1} + \cdots + c_n x_n = 0.\]
    Thus $x_1, \ldots, x_n$ is linearly dependent.
\end{proof}


\begin{tcolorbox}[title=Problem 7, breakable]
    If $T : V \longrightarrow W$ is an injective 
    linear mapping and if $x_1, \ldots, x_n$ are vectors 
    in $V$ such that $Tx_1, \ldots, Tx_n$ are linearly 
    dependent in $W$, then $x_1, \ldots, x_n$ are 
    linearly dependent in $V$.
\end{tcolorbox}

\begin{proof}
    Suppose $T : V \longrightarrow W$ is an injective 
    linear mapping and $Tx_1, \ldots, Tx_n$ are linearly 
    dependent in $W$. We have scalars $c_1, \ldots, c_n$ not all zero such that
    \[c_1 Tx_1 + \cdots + c_n Tx_n = T(c_1 x_1 + \cdots + c_n x_n) = 0.\]
    Since $T$ is injective we have
    \[c_1 x_1 + \cdots + c_n x_n = 0.\]
    Thus $x_1, \ldots, x_n$ are linearly dependent.
\end{proof}

\subsection{Linear Independence}

\begin{tcolorbox}[title=Problem 1, breakable]
    Show that the vectors $(2, -1, 1), (-1, 2, 1), (1, 1, 1)$
    in $\mathbb{R}^3$ are linearly independent.
\end{tcolorbox}

\begin{proof}
    Suppose there exist scalars not all zero $c_1, c_2, c_3$ such that 
    \[c_1 (2, -1, 1) + c_2 (-1, 2, 1) + c_3 (1, 1, 1) = 0.\]
    Wlog suppose $c_1 \ne 0$ so we have 
    \[(2, -1, 1) = -\frac{c_2}{c_1} (-1, 2, 1) - \frac{c_3}{c_1}(1, 1, 1).\]
    From this we get the following equations
    \begin{enumerate}
        \item $2 = \frac{c_2}{c_1} - \frac{c_3}{c_1}$
        \item $-1 = -2\frac{c_2}{c_1} - \frac{c_3}{c_1}$
        \item $1 = -\frac{c_2}{c_1} - \frac{c_3}{c_1}$
    \end{enumerate}
    Let $a = \frac{c_2}{c_1}$ and $b = \frac{c_3}{c_1}$. Then the system becomes
    \begin{align*}
        2 &= a - b \\
        -1 &= -2a - b \\
        1 &= -a - b.
    \end{align*}
    From the first equation, $a = 2 + b$. Substituting into the third gives
    \[
        1 = -(2 + b) - b = -2 - 2b,
    \]
    so $b = -\frac{3}{2}$ and thus $a = \frac{1}{2}$. Substituting into the second equation
    \[
        -1 = -2\left(\frac{1}{2}\right) - \left(-\frac{3}{2}\right) = -1 + \frac{3}{2} = \frac{1}{2},
    \]
    a contradiction.
\end{proof}

\begin{tcolorbox}[title=Problem 2, breakable]
    If $x_1, \ldots, x_n, x_{n + 1}$ are linearly 
    independent, then so are $x_1, \ldots, x_n$.
\end{tcolorbox}

\begin{proof}
    Suppose $x_1, \ldots, x_n, x_{n + 1}$ are linearly 
    independent.
    For contradiction, suppose $x_1, \ldots, x_n$
    are dependent.
    Then for scalars $c_1, \ldots, c_n$ not all zero we have 
    \[\sum_{i=1}^n c_i x_i = 0 \text{ so } \sum_{i=1}^n c_i x_i + 0\cdot x_{n + 1} = 0.\]
    Contradicting that $x_1, \ldots, x_n, x_{n + 1}$ are linearly independent.
\end{proof}


\begin{tcolorbox}[title=Problem 3, breakable]
    If $x_1, \ldots, x_n$ are linearly independent,
    then the following implication is true:
    (*) $a_1 x_1 + \cdots + a_n x_n = b_1 x_1 + \cdots + b_n x_n \implies a_i = b_i$ for all $i$.
    Conversely, the validity of (*) implies that $x_1, \ldots, x_n$ are lienarly independent.
\end{tcolorbox}

\begin{proof}
    \underline{\textbf{($\longrightarrow$)}}
    Suppose $x_1, \ldots, x_n$ are linearly independent.
    Suppose 
    \[\sum a_i x_i = \sum b_i x_i.\]
    Then
    \[\sum (a_i - b_i)x_i = 0.\]
    Since the vectors are linearly independent it follows that 
    \[a_i - b_i = 0 \text{ for all } i,\]
    thus $a_i = b_i$ for all $i$.

    \underline{\textbf{($\longleftarrow$)}}
    Suppose (*) holds. Now suppose
    \[\sum a_i x_i = 0.\]
    Then
    \[\sum a_i x_i = \sum 0 \cdot x_i.\]
    By (*) it follows that $a_i = 0$ for all $i$. Thus $x_1, \ldots, x_n$
    are linearly independent.
\end{proof}

\begin{tcolorbox}[title=Problem 9, breakable]
    In the vector space $\mathcal{F}(\mathbb{R})$ of all 
    functions $\mathbb{R} \longrightarrow \mathbb{R}$ (Example 1.3.4),
    $\sin$ and $\cos$ are linearly independent; so are the functions 
    $\sin, \cos, p$ where $p$ is any nonzero polynomial function 
    (Hint: Differentiate.); so are $\sin, \cos, p, exp$ (the base $e$ exponential function).
\end{tcolorbox}

\begin{proof}
    Suppose $\sin$ and $\cos$ are linearly dependent
    thus there exists scalars $c_1, c_2$ not both zero such that 
    \[
        c_1 \cos(x) + c_2 \sin(x) = 0.
    \]
    At $x = 0$ we have
    \[
        c_1 \cos(0) + c_2 \sin (0) = 0 \implies c_1 = 0.
    \]
    But then at $x = \pi / 2$ we have 
    \[
        c_1 \cos(\pi/2) + c_2 \sin (\pi/2) = 0 \implies c_2 = 0.
    \]
    Thus contradicting that $c_1, c_2$ are not both zero.

    Suppose $\sin, \cos,$ and $p$ are linearly dependent where $p$ is any nonzero polynomial function.
    There exists scalars $c_1, c_2, c_3$ not all zero such that 
    \[
        c_1 \cos(x) + c_2 \sin(x) + c_3 p(x) = 0.
    \]
    If $c_3 = 0$ then we have the previous case.
    Therefore suppose $c_3 \ne 0$ and we find 
    \[
        p(x) = -\frac{c_1}{c_3} \cos(x) - \frac{c_2}{c_3} \sin(x).
    \]
    If $p$ is not constant, the rhs is bounded, while a nonzero polynomial is unbounded, giving a contradiction.
    If $p$ is constant, then the rhs would have to be constant, but a nontrivial linear combination of $\sin$ and $\cos$ is not constant, also giving a contradiction.

    Suppose $\sin, \cos, \exp$ and $p$ are linearly dependent where $p$ is any nonzero polynomial function.
    There exists scalars $c_1, c_2, c_3, c_4$ not all zero such that 
    \[
        c_1 \cos(x) + c_2 \sin(x) + c_3 p(x) + c_4 e^x = 0.
    \]
    If $c_4 = 0$ we have the previous case.
    Therefore suppose $c_4 \ne 0$.
    Let $p$ have degree $n$. Differentiating $n+1$ times gives
    \[
        A \cos(x) + B \sin(x) + C e^x = 0
    \]
    for some constants $A,B,C$ with $C \ne 0$.
    Solving for $e^x$ we obtain
    \[
        e^x = -\frac{A}{C} \cos(x) - \frac{B}{C} \sin(x).
    \]
    The rhs is bounded, while $e^x$ is unbounded as $x \to \infty$,
        giving a contradiction.
\end{proof}


\begin{tcolorbox}[title=Problem 12, breakable]
    Let $V = \mathbb{C}^2$ and regard $V$ as a real vector space by restriction
    of scalars (1.3, Exercise 3). Let $x_1 = (1, i), x_2 = (i, -1)$.
    Show that $x_1, x_2$ are linearly independent in $V_{\mathbb{R}}$ but 
    linearly dependent in $V_{\mathbb{C}}$.
\end{tcolorbox}

\begin{proof}
    Let $c_1 = -i$ and $c_2 = 1$. Then
    \[
        c_1 x_1 + c_2 x_2 = (-i)(1,i) + 1(i,-1)
        = (-i + i,\,-i^2 - 1)
        = (0,0).
    \]
    Thus $x_1, x_2$ are linearly dependent in $V_{\mathbb{C}}$.

    Now suppose $a_1 x_1 + a_2 x_2 = 0$ with $a_1, a_2 \in \mathbb{R}$.
    Then
    \[
        a_1 (1, i) + a_2 (i, -1) = (a_1 + i a_2,\, i a_1 - a_2) = 0
    \]
    so
    \[
        a_1 + i a_2 = 0, \quad i a_1 - a_2 = 0.
    \]
    From $a_1 + i a_2 = 0$, we get $a_1 = -i a_2$. Since $a_1 \in \mathbb{R}$,
    we have $a_2 = 0$ so $a_1 = 0$ which is a contradiction.
    Therefore $x_1, x_2$ are linearly independent in $V_{\mathbb{R}}$.
\end{proof}

\begin{tcolorbox}[title=Problem 14, breakable]

\begin{proof}
    Let $p$ be a polynomial of degree $n$.
    Suppose $I, D, D^2, \ldots, D^n$ are linearly dependent.
    To simplify notation, let $I = D^0$.
    Let $c_0, c_1, \ldots, c_n$ be scalars not all zero such that

\end{proof}


\begin{tcolorbox}[title=Problem 15, breakable]
    A set $A$ of vectors is said to be linearly independent
    if every finite list $x_1, \ldots, x_n$ of distinct 
    vectors in $A$ is linearly independent (in other words,
    $A$ is not linearly dependent in the sense of 3.2, Exercise 8)
    \begin{enumerate}
        \item $A$ is linearly independent if and only if no vector $A$ is 
        linear combination of remaining vectors of $A$, that is, for every 
        vector $x \in A$, $x \ne [A - \{x\}]$.
        \item The empty set $\emptyset$ is linearly independent `by default'.
        {Hint: Independence of $A$ means $x \in A \implies x \notin [A - \{x\}]$
        Remember `vacuous implication' (Appendix A.1.5)?}
        \item If $A$ is linearly independent and $B \subset A$, then $B$ is linearly independent.
        \item If $T : V \longrightarrow W$ is an injective linear mapping and $A$
        is linearly independent subset of $V$, then $T(A)$ is linearly independent subset of $W$.
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    \underbar{\textbf{(1)}} \\
    \underbar{($\longrightarrow$)}
    Suppose $A$ is linearly independent. Let $x \in A$. If $x \in [A - \{x\}]$, then
    \[
        x = \sum c_i x_i \quad (x_i \in A \setminus \{x\}),
    \]
    so
    \[
        x - \sum c_i x_i = 0,
    \]
    a linear dependence in $A$ which is a contradiction. Thus $x \notin [A - \{x\}]$.

    \underbar{\textbf{($\longleftarrow$)}}\
    Suppose no $x \in A$ satisfies $x \in [A - \{x\}]$. If
    \[
        \sum c_i x_i = 0
    \]
    with some $c_j \ne 0$, then
    \[
        x_j = -\sum_{i \ne j} \frac{c_i}{c_j} x_i \in [A - \{x_j\}],
    \]
    a contradiction. Thus $A$ is linearly independent.

    \medskip
    \underbar{\textbf{(2)}} \\
    The condition is vacuously true for $A = \emptyset$, so $\emptyset$ is linearly independent.

    \medskip
    \underbar{\textbf{(3)}} \\
    Let $B \subset A$. Any finite list in $B$ is also a finite list in $A$, 
        so it is linearly independent since $A$ is. Thus $B$ is linearly independent.

    \medskip
    \underbar{\textbf{(4)}}\\
    Let $T: V \to W$ be injective and let $A$ be linearly independent. If
    \[
    \sum c_i T(x_i) = 0,
    \]
    then
    \[
    T\left(\sum c_i x_i\right) = 0.
    \]
    Since $T$ is injective, $\sum c_i x_i = 0$, so all $c_i = 0$. 
    Thus $T(A)$ is linearly independent.
\end{proof}

\begin{tcolorbox}[title=Problem 3, breakable]
    Let $V$ and $W$ be vector spaces over $F$,
    $V \times W$ the product vector space 
    (1.3.11). Prove: $V \times W$ is finitely generated 
    if and only if both $V$ and $W$ are finitely generated.
    {Hint: Remark 2.3.8 3.4.12 and 3.1, Exercise 6.}
\end{tcolorbox}

\begin{proof}
    \underline{\textbf{($\longrightarrow$)}}
    Suppose $V \times W$ is finitely generated.
    So there is a finite list of vectors 
    $(v_1,w_1), \ldots, (v_n,w_n) \in V \times W$ such that 
    \[
    V \times W = [\{(v_1,w_1), \ldots, (v_n,w_n)\}].
    \]
    Let $v \in V$. Then $(v,0) \in V \times W$ so 
    $(v,0)$ is a linear combination of 
    $(v_1,w_1), \ldots, (v_n,w_n)$. Thus 
    $v$ is a linear combination of $v_1,\ldots,v_n$ and thus
    \[
    V=[\{v_1,\ldots,v_n\}].
    \]
    Similarly
    \[
    W=[\{w_1,\ldots,w_n\}].
    \]
    Thus both $V$ and $W$ are finitely generated.

    \underline{\textbf{($\longleftarrow$)}}
    Suppose both $V$ and $W$ are finitely generated. Then 
    \[
    V = [\{v_1, \ldots, v_n\}] \text{ and } W = [\{w_1, \ldots, w_m\}].
    \]
    We have 
    \[
    V \times W = [\{v_1, \ldots, v_n\}] \times [\{w_1, \ldots, w_m\}].
    \]
    Then
    \[
    V \times W
    =
    [\{(v_1,0),\ldots,(v_n,0),(0,w_1),\ldots,(0,w_m)\}],
    \]
    thus $V \times W$ is finitely generated.
\end{proof}


\begin{tcolorbox}[title=Problem 4, breakable]
    Let $V$ be a vector space of all function 
    $\mathbb{R} \longrightarrow \mathbb{R}$ having 
    derivatives of all orders (see Example 2.3.8) and let 
    \[M = \{f \in V : f'' = f\},\]
    where $f''$ is the second derivative of $f$.
    Prove that $M$ is the linear subspace of $V$ generated 
    by the two functions $t \longrightarrow e^t$ and $t \longrightarrow e^{-t}$.
    {Hint: if $f'' = f$, look at $f = \frac{1}{2}(f + f') + \frac{1}{2}(f - f')$.}
\end{tcolorbox}

\begin{proof}
    It is clear that $A e^t + B e^{-t} \in M$.
    Let $f$ be an arbitrary element in $M$.
    Thus 
    \[f'' = f.\]
    Thus 
    \[f = \frac{1}{2}(f + f') + \frac{1}{2}(f - f').\]
    Let $g = \frac{1}{2}(f + f')$ and $h =  \frac{1}{2}(f - f')$.
    Then $f = g + h$.
    Then 
    \[g' = \frac{1}{2}(f' + f'') = \frac{1}{2}(f' + f) = g,\]
    so $g = A e^t$ for some constant $A$.
    Similarly
    \[h' = \frac{1}{2}(f' - f'') = \frac{1}{2}(f' - f) = -h,\]
    so $h = B e^{-t}$ for some constant $B$.
    Thus $f = A e^t + B e^{-t}$.
\end{proof}

\begin{tcolorbox}[title=Problem 5, breakable]
    With notations as in Exercise 4, $M$ is also generated 
    by the function $sinh$, $cosh$.
\end{tcolorbox}

\begin{proof}
    Simply note 
    \[\sinh(t) = \frac{e^t - e^{-t}}{2} \text{ and } \cosh(t) = \frac{e^t + e^{-t}}{2}.\]
\end{proof}


\begin{tcolorbox}[title=Problem 6, breakable]
    With $V$ as in Exercise 4, let 
    \[N = \{f \in V \mid f'' = -f\}.\]
    Prove that $N$ is generated by the function $\sin, \cos$.
    {Hint: If $f'' = -f,$ calculate the derivative of $f \sin + f' \cos$ 
    and of $f\cos - f'\sin$.}
\end{tcolorbox}

\begin{proof}
    It is clear that $A\sin t + B\cos t \in N$.
    Let $f$ be an arbitrary element in $N$. Thus $f'' = -f$.
    Now
    \[
    \frac{d}{dt}(f \sin t + f' \cos t)
    = f' \sin t + f \cos t + f'' \cos t - f' \sin t
    = f \cos t + f'' \cos t = 0.
    \]
    Thus $f \sin t + f' \cos t = A$ for some constant $A$.
    Then
    \[
    \frac{d}{dt}(f \cos t - f' \sin t)
    = f' \cos t - f \sin t - f'' \sin t - f' \cos t
    = -f \sin t - f'' \sin t = 0.
    \]
    Thus $f \cos t - f' \sin t = B$ for some constant $B$.
    Solving these two equations gives $f = A \sin t + B \cos t$.
\end{proof}

\begin{tcolorbox}[title=Problem 7, breakable]
    If $V$ is a vector space and $M$ is linear subspace 
    such that both $M$ and $V/M$ are finitely generated,
    then $V$ is finitely generated. More precisely,
    if $x_1, \ldots, x_m$ generate $M$ and if 
    $Q y_1, \ldots, Q y_r$ generate $V/M$ (where 
    $Q : V \longrightarrow V/M$ is the quotient mapping),
    then $V$ is generated by $x_1, \ldots, x_m, y_1, \ldots, y_r$.
\end{tcolorbox}

\begin{proof}
    Suppose $x_1, \ldots, x_m$ generate $M$ and 
    $Q y_1, \ldots, Q y_r$ generate $V/M$.
    Let $v$ be an arbitrary element in $V$.
    Since $Qy_1, \ldots, Qy_r$ generate $V/M$, there exist scalars 
    $c_1, \ldots, c_r$ such that 
    \[Qv = \sum c_i Qy_i.\]
    Thus
    \[Qv = Q\left(\sum c_i y_i\right) \implies v - \sum c_i y_i \in M.\]
    Thus 
    \[\sum c_i y_i = v + m \text{ where } m \in M.\]
    Since $M$ is finitely generated there exist 
    $x_1, \ldots, x_m$ and scalars $c'_1, \ldots, c'_m$ such that 
    \[m = \sum c'_i x_i.\]
    So 
    \[\sum c_i y_i = v + \sum c'_i x_i \implies v = \sum c_i y_i - \sum c'_i x_i.\]
\end{proof}

\begin{tcolorbox}[title=Problem 1, breakable]
    Prove that $(1, 2, 3), (2, 1, 0), (-3, 2, 0)$
    is a basis of $\mathbb{R}^3$. Then find a shorter proof.
\end{tcolorbox}

\begin{proof}
    Let $(a, b, c) \in \mathbb{R}^3$.
    We require $c_1, c_2, c_3$ such that 
    \[c_1 (1, 2, 3) + c_2 (2, 1, 0) + c_3 (-3, 2, 0) = (a, b, c).\]
    Thus 
    \[(c_1, 2 c_1, 3 c_1) + (2 c_2, c_2, 0) + (-3 c_3, 2 c_3, 0) = (a, b, c).\]
    From this we get three equations 
    \[c_1 + 2 c_2 - 3 c_3 = a, \quad 2 c_1 + c_2 + 2 c_3 = b, \quad 3 c_1 = c.\]
    So we have 
    \[c_1 = \frac{c}{3}, \quad c_2 = \frac{2a + 3b - 8c}{21}, \quad c_3 = \frac{2b - a - c}{7}.\]
    Verifying,
    \[
    c_1 (1, 2, 3) + c_2 (2, 1, 0) + c_3 (-3, 2, 0)
    =
    \frac{c}{3} (1, 2, 3)
    + \frac{2a + 3b - 8c}{21} (2, 1, 0)
    + \frac{2b - a - c}{7} (-3, 2, 0)
    = (a, b, c).
    \]
    Therefore, $(1, 2, 3), (2, 1, 0), (-3, 2, 0)$ generate $\mathbb{R}^3$.
    Now suppose there exists $c_1, c_2, c_3$ not all zero such that 
    \[c_1 (1, 2, 3) + c_2 (2, 1, 0) + c_3 (-3, 2, 0) = 0.\]
    We get three equations
    \[
    c_1 + 2c_2 - 3c_3 = 0, \quad
    2c_1 + c_2 + 2c_3 = 0, \quad
    3c_1 = 0.
    \]
    From $3c_1 = 0$ we have $c_1 = 0$. Substituting into the first two equations gives
    \[
    2c_2 - 3c_3 = 0, \quad c_2 + 2c_3 = 0.
    \]
    From the second equation $c_2 = -2c_3$. Substituting into the first gives
    \[
    2(-2c_3) - 3c_3 = -7c_3 = 0,
    \]
    so $c_3 = 0$ thus $c_2 = 0$. Therefore
    \[
    c_1 = c_2 = c_3 = 0.
    \]
    Thus the vectors are linearly independent.
    Since $(1, 2, 3), (2, 1, 0), (-3, 2, 0)$ are generating and linearly independent
        they form a basis of $\mathbb{R}^3$.
\end{proof}

\begin{tcolorbox}[title=Problem 2, breakable]
    Let $F$ be any field.
    Explain why $dim_F F^n = n$.
\end{tcolorbox}

\begin{proof}
    We know that $\{e_1, \ldots, e_n\}$ forms a basis of $F^n$ and $|\{e_1, \ldots, e_n\}| = n$.

\end{proof}

\begin{tcolorbox}[title=Problem 3, breakable]
    Prove that $F^m \cong F^n$
    if and only if $m = n$.
\end{tcolorbox}

\begin{proof}
    \underbar{\textbf{($\longrightarrow$)}}
    Suppose $F^m \cong F^n$.
    There exists a bijective linear mapping $L : F^m \longrightarrow F^n$.
    Then $L(\{e_1, \ldots, e_m\}) = \{L e_1, \ldots, L e_m\}$.
    Since $L$ is surjective, $\{L e_1, \ldots, L e_m\}$ generates $F^n$, and since $L$ is injective, these vectors are linearly independent.
    Thus $m = n$.

    \underbar{\textbf{($\longleftarrow$)}}
    Suppose $m = n$.
    Consider the bijective linear mapping $L : F^n \longrightarrow F^n$ defined by $L(e_i) = e_i$.
    Then $L$ is an isomorphism, so $F^m \cong F^n$.
\end{proof}

\begin{tcolorbox}[title=Problem 4, breakable]
    Let $V$ be a vector space, $V \ne \{0\}$.
    Prove that $V$ is finitely generated if 
    and only if $V \cong F^n$ for some 
    positive integer $n$; if $V \cong F^n$
    then $n = dim_F V$ (in particular, $n$ is unique).
\end{tcolorbox}

\begin{proof}
    \underbar{\textbf{($\longrightarrow$)}}
    Suppose $V$ is finitely generated.
    Say $\{x_1, \ldots, x_n\}$ is a basis for $V$.
    Consider the bijective linear mapping $L: F^n \longrightarrow V$ defined 
    \[L(a_1, \ldots, a_n) = \sum a_i x_i.\]
    Thus $V \cong F^n$.

    \underbar{\textbf{($\longleftarrow$)}}
    Suppose $V \cong F^n$ for some positive integer $n$.
    Thus there exists a bijective linear mapping $L: F^n \longrightarrow V$.
    Let $\{e_1, \ldots, e_n\}$ be the standard basis of $F^n$.
    Then $\{L(e_1), \ldots, L(e_n)\}$ spans $V$ and is linearly independent thus is a basis of $V$.
    Therefore $V$ is finitely generated.
\end{proof}

\begin{tcolorbox}[title=Problem 5, breakable]
    Let $\alpha$ be a real number. Prove that the vectors 
    \[u = (\cos \alpha, \sin \alpha), v = (-\sin \alpha, \cos \alpha),\]
    are a basis of $\mathbb{R}^2$
\end{tcolorbox}

\begin{proof}
    Suppose there exists $c$ such that 
    \[(\cos \alpha, \sin \alpha) = c(-\sin \alpha, \cos \alpha).\]
    We get two equations 
    \[\cos\alpha = -c \sin \alpha \text{ and } \sin \alpha = c \cos \alpha.\]

    Multiplying these equations gives
    \[
    \cos\alpha \sin\alpha = -c^2 \sin\alpha \cos\alpha.
    \]
    Thus
    \[
    (1+c^2)\sin\alpha\cos\alpha = 0.
    \]
    Since $1+c^2 \neq 0$ we have $\sin\alpha =0$ or $\cos\alpha=0$.
    In either case both give $\sin\alpha=0$ and $\cos\alpha=0$,
    a contradiction.
    Since $\dim \mathbb{R}^2 = 2$ and $\{u, v\}$ 
        contains two linearly independent vectors in $\mathbb{R}^2$ it forms a basis of $\mathbb{R}^2$.
\end{proof}

\begin{tcolorbox}[title=Problem 9, breakable]
    Let $V$ be a finite-dimensional vector space and let $M$ be a linear subspace of $V$. Prove:
    \begin{enumerate}
        \item $V/M$ is finite-dimensional, and $dim(V/M) \le dim V$;
        \item $dim(V/M) = dim V \iff M = \{0\}$;
        \item $dim(V/M) = 0 \iff M = V$.
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    By 3.5.14, $M$ is finite-dimensional.
    Furthermore, by 3.5.16 we can write
    \[[\{x_1, \ldots, x_r, x_{r+1}, \ldots, x_n\}] = V \text{ with } [\{x_{r+1}, \ldots, x_n\}] = M.\]
    By 3.4.8, $Qx_1, \ldots, Qx_r$ generate $V/M$, 
        so $V/M$ is finite-dimensional with $\dim(V/M) \le r \le n = \dim V$.
\end{proof}

\begin{proof}
    \underline{\textbf{($\longrightarrow$)}}
    Suppose $\dim(V/M) = \dim V = n$.
    Suppose $Qx_1, \ldots, Qx_r$ and $x_{r+1}, \ldots, x_n$ generate $V/M$ and $M$ respectively.
    We know that $x_1, \ldots, x_r, x_{r+1}, \ldots, x_n$ generate $V$ by Problem 7.
    Then $\dim V = r + \dim M = n$, thus $\dim M = n - r$. But $\dim(V/M) = r = n$,
    so $\dim M = 0 \implies M = \{0\}$.

    \underline{\textbf{($\longleftarrow$)}}
    Suppose $M = \{0\}$.
    Suppose $Qx_1, \ldots, Qx_r$ and $x_{r+1}, \ldots, x_n$ generate $V/M$ and $M$ respectively.
    We know that $x_1, \ldots, x_r, x_{r+1}, \ldots, x_n$ generate $V$ by Problem 7.
    But $\dim M = 0$, so there are no $x_{r+1}, \ldots, x_n$, thus $\dim(V/M) = r = n = \dim V$.
\end{proof}

\begin{proof}
    \underline{\textbf{($\longrightarrow$)}}
    Suppose $\dim(V/M) = 0$.
    Suppose $Qx_1, \ldots, Qx_r$ and $x_{r+1}, \ldots, x_n$ generate $V/M$ and $M$ respectively.
    Let $v \in V$. Then
    \[v = \sum_{i=1}^r a_i x_i + \sum_{i=r+1}^n b_i x_i.\]
    But since $\dim(V/M) = 0$ there are no terms in $\sum_{i=1}^r a_i x_i$.
    Thus $v = \sum_{i=r+1}^n b_i x_i$ where each $x_i \in M$, thus $v \in M$.
    So $V \subset M$. We know $M \subset V$, thus $V = M$.

    \underline{\textbf{($\longleftarrow$)}}
    Suppose $M = V$.
    Suppose $Qx_1, \ldots, Qx_r$ and $x_{r+1}, \ldots, x_n$ generate $V/M$ and $M$ respectively.
    But since $M = V$, every $x_i \in M$, so $Qx_i = 0$ for all $i$.
    Thus $V/M$ is generated by $\{0\}$, thus $\dim(V/M) = 0$.
\end{proof}

\begin{tcolorbox}[title=Problem 11, breakable]
    Let $x_1, \ldots, x_n$ be a basis of $V$ and let $1 \le m < n$.
    Let $M$ be the linear subspace of $V$ generated by $x_1, \ldots, x_m$, 
    and $N$ the linear subspace generated by $x_{m + 1}, \ldots, x_n$. Prove that 
    $V = M \oplus N$ (in the sense of 1.6, Exercise 12).
\end{tcolorbox}

\begin{proof}
    Let $x \in V$. We can write 
    \[x = \sum_{i=1}^n c_i x_i,\]
    for suitable $c_i \in F$.
    So 
    \[
    x = \sum_{i=1}^m c_i x_i + \sum_{i=m+1}^n c_i x_i \in M + N.
    \]
    Thus $V = M + N$.

    It remains to show that $M \cap N = \{0\}$.
    Suppose $x \in  M \cap N$ such that 
    \[
    x=\sum_{i=1}^m a_i x_i
    \]
    and
    \[
    x=\sum_{i=m+1}^n b_i x_i.
    \]
    Then
    \[
    \sum_{i=1}^m a_i x_i-\sum_{i=m+1}^n b_i x_i=0.
    \]
    Since $x_1,\ldots,x_n$ is a basis, it is linearly independent, so all
    coefficients are zero. Thus $a_i=0$ and $b_i=0$ for all $i$, and thus
    $x=0$.
    Therefore $M \cap N=\{0\}$, so $V=M\oplus N$.
\end{proof}


\begin{tcolorbox}[title=Problem 12, breakable]
    Let $V$  be a vector space over $F$,
    let $x_1, \ldots, x_n$ be nonzero vectors in $V$,
    and let $M_i = F x_i$ for $i = 1, \ldots, n$.
    Prove that the following two conditions are equivalent:
    \begin{enumerate}
        \item $x_1, \ldots, x_n$ is a basis of $V$;
        \item $V = M_1 \oplus \cdots \oplus M_n$ (in the sense of 3.3, Exercise 20).
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    \underbar{\textbf{($\longrightarrow$)}}
    Suppose $x_1, \ldots, x_n$ is a basis of $V$.
    Let $M_i = [\{x_i\}]$. We have mapping $T : M_1 \times \cdots \times M_n \longrightarrow V$ defined 
    \[T(v_1, \ldots, v_n) = \sum v_i.\]

    Suppose $v \in V$ we have 
    \[v = \sum c_i x_i,\]
    for appropriate $c_1, \ldots, c_n \in F$, so $c_i x_i \in M_i$.
    Since $x_1, \ldots, x_n$ are linearly independent
        we have $\ker T = \{0\}$. Thus $M_1, \ldots, M_n$ are linearly independent.
    Thus $v \in  M_1 \oplus \cdots \oplus M_n$.

    Since $M_i \subseteq V$ for each $i$, we have 
    $M_1 + \cdots + M_n \subseteq V$.

    Thus $V = M_1 \oplus \cdots \oplus M_n$.

    \underbar{\textbf{($\longleftarrow$)}}
    Suppose $V = M_1 \oplus \cdots \oplus M_n$.
    Now, suppose $v \in V$.
    Then $v = v_1 + \cdots + v_n$ with $v_i \in M_i$.
    Then for scalars $c_1, \ldots, c_n \in F$ we have 
    $v_i = c_i x_i$.
    Clearly the $x_1, \ldots, x_n$ are generating.
    By definition of $\oplus$ the $x_i$ are linearly independent.
    Thus $x_1, \ldots, x_n$ is a basis of $V$.
\end{proof}

\begin{tcolorbox}[title=Problem 17, breakable]
    Prove that if $B$ is a basis of $V$ (in the sense 
    of Exercise 16), then $B$ is `minimal-generating' and `maximal-independent' in the following sense:
    \begin{enumerate}
        \item If $A$ is a proper subset of $B$ (that is, $A \subset B$ and $A \ne B$) then $A$ is not generating;
        \item If $B \subset C \subset V$ and $B$ is a proper subset of $C$, then $C$ is not linearly independent.
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    Suppose $B$ is a basis of $V$.
    Now, suppose $A$ is a proper subset of $B$.
    For contradiction, suppose $[A] = V$.
    Let $x \in B - A$. Since $x \in V$ we have $x = \sum c_i a_i$
    for some $c_i \in F$ and $a_i \in A$.
    But since $A \subset B$ we have $x = \sum c_i b_i$ where $b_i \in B$.
    This contradicts that $B$ is linearly independent.

    Now, suppose $B \subset C \subset V$ and $B$ is a proper subset of $C$.
    Let $x \in C - B$. Since $x \in V$ we have $x = \sum c_i b_i$ where $b_i \in B$.
    But since $B \subset C$ we have $b_i \in C$, so $x$ is a linear combination of elements of $C$.
    Thus $C$ is not linearly independent.
\end{proof}


\begin{tcolorbox}[title=Problem 18, breakable]
    Let $\mathcal{D}$ be the vector space of real polynomial functions (Example 1.3.5).
    For $n = 0, 1, 2, \ldots$ let $p_n$ be the monomial function 
    $p_n(t) = t^n$, and let $B = \{p_0, p_1, p_2, \ldots\}$. Prove that $B$ 
    is a basis of $\mathcal{D}$ in the sense of Exercise 16.
\end{tcolorbox}

\begin{proof}
    Let $f$ be an arbitrary polynomial in $\mathcal{D}$.
    Let $\deg(f) = m$. By definition $f$ can be written 
    \[f = \sum_{i=0}^m c_i t^i = \sum_{i=0}^m c_i p_i(t) \in [B].\]
    Thus $B$ is generating.

    Now suppose $C \subset B$ and $C \ne B$.
    Let $g \in B - C$. Then $g = p_k$ for some $k$, and $p_k \notin C$.
    Since every element of $[C]$ is a linear combination of elements of $C$,
    it cannot equal $p_k$. Thus $[C] \ne [B] = \mathcal{D}$, so $C$ is not generating.

    It follows that $B$ is a basis of $\mathcal{D}$.
\end{proof}

\begin{tcolorbox}[title=Problem 19, breakable]
    Let $V$ be the vector space of finitely nonzero 
    sequences of real numbers (1.3.9). Find a 
    basis $B$ of $V$ (in the sense of Exercise 16).
\end{tcolorbox}

\begin{proof}
    For $n = 0, 1, 2, \ldots$ let $p_n$ be the sequence
        $p_n = (0, \ldots, 0, 1, 0, \ldots)$ where the $1$ is in the $n$-th position, and let 
        $B = \{p_0, p_1, p_2, \ldots\}$.

    Let $x$ be an arbitrary element of $V$. Since $x$ is a finitely nonzero sequence, we have
    \[x = (c_0, c_1, \ldots, c_m, 0, 0, \ldots).\]
    Then
    \[x = \sum_{i=0}^m c_i p_i \in [B].\]
    Thus $B$ is generating.

    Suppose $C \subset B$ and $C \ne B$.
    Let $g \in B - C$. Then $g = p_k$ for some $k$.
    Since every element of $[C]$ has zero in the $k$-th position,
    it cannot equal $p_k$. Thus $[C] \ne V$, so $C$ is not generating.

    It follows that $B$ is a basis of $V$.
\end{proof}


\begin{tcolorbox}[title=Problem 20, breakable]
    Let $V$ be a finite-dimensional complex vector space.
    Regard $V$ also as a real vector space by restriction of scalars (1.3, Exercise 3).
    Prove that $dim_{\mathbb{R}} V = 2 \cdot dim_{\mathbb{C}} V$.
\end{tcolorbox}

\begin{proof}
    Let $\{v_1, \ldots, v_n\}$ be a basis of $V$ over $\mathbb{C}$.
    Let $v \in V$ such that
    \[
    v = \sum_{k=1}^n z_k v_k
    \]
    where $z_k \in \mathbb{C}$.
    We can write $z_k = a_k + b_k i$ for $a_k, b_k \in \mathbb{R}$.
    Then
    \[
    v = \sum_{k=1}^n (a_k + b_k i)v_k
    = \sum_{k=1}^n a_k v_k + \sum_{k=1}^n b_k (i v_k).
    \]
    Consider the set
    \[
    B = \{v_1, iv_1, \ldots, v_n, iv_n\}.
    \]
    Now, consider the bijection $T : \mathbb{R}^{2n} \to V$ defined 
    \[
    T(a_1, b_1, \ldots, a_n, b_n)
    = \sum_{k=1}^n a_k v_k + \sum_{k=1}^n b_k (i v_k).
    \]
    It follows that $B$ is a basis of $V$ over $\mathbb{R}$.
    Therefore
    \[
    \dim_{\mathbb{R}} V = 2 \cdot \dim_{\mathbb{C}} V.
    \]
\end{proof}

\begin{tcolorbox}[title=Problem 21, breakable]
    If $V$ is a finite-dimensional real vector space and $W$
    is its complexification (1.3, Exercise 4), 
    prove that $dim_C W = dim_R V$.
\end{tcolorbox}

\begin{proof}
    Let $\{v_1, \ldots, v_n\}$ be a basis of $V$ over $\mathbb{R}$.
    Since $W$ is the complexification of $V$, every element of $W$ can be written as
    \[
    w = \sum_{i=1}^n c_i v_i
    \]
    where $c_1, \ldots, c_n \in \mathbb{C}$.
    Consider the map $T : \mathbb{C}^n \longrightarrow W$ defined by
    \[
    T(c_1, \ldots, c_n) = \sum_{i=1}^n c_i v_i.
    \]
    $T$ is a bijection so $\{v_1, \ldots, v_n\}$ is a basis of $W$ over $\mathbb{C}$.
    Therefore
    \[
    \dim_{\mathbb{C}} W = n = \dim_{\mathbb{R}} V.
    \]
\end{proof}


\begin{tcolorbox}[title=Problem 22, breakable]
    Let $x_1, \ldots, x_n$ be vectors in $\mathbb{R}^n$.
    Prove that $x_1, \ldots, x_n$ is a basis of the real vector 
    space $\mathbb{R}^n$ if and only if it is a basis of the complex 
    vector space $\mathbb{C}^n$.
\end{tcolorbox}

\begin{proof}
    \underbar{\textbf{($\longrightarrow$)}}
    Suppose $x_1, \ldots, x_n$ is a basis of $\mathbb{R}^n$.
    Let $v \in \mathbb{C}^n$. Then $v = u + iw$ where $u, w \in \mathbb{R}^n$.
    Since $x_1, \ldots, x_n$ span $\mathbb{R}^n$ we can write $u = \sum a_i x_i$ and $w = \sum b_i x_i$ with $a_i, b_i \in \mathbb{R}$, so
    \[v = \sum a_i x_i + i \sum b_i x_i = \sum (a_i + b_i i) x_i,\]
    thus $x_1, \ldots, x_n$ span $\mathbb{C}^n$.
    Now let $\sum_{i=1}^n c_i x_i = 0$ with $c_i \in \mathbb{C}$. Writing $c_i = a_i + b_i i$,
    \[\sum_{i=1}^n a_i x_i + i \sum_{i=1}^n b_i x_i = 0 \implies \sum_{i=1}^n a_i x_i = 0 \text{ and } \sum_{i=1}^n b_i x_i = 0.\]
    Since $x_1, \ldots, x_n$ are linearly independent over $\mathbb{R}$, all $a_i = b_i = 0$, thus all $c_i = 0$.
    Thus $x_1, \ldots, x_n$ is a basis of $\mathbb{C}^n$.

    \underbar{\textbf{($\longleftarrow$)}}
    Suppose $x_1, \ldots, x_n$ is a basis of $\mathbb{C}^n$.
    Let $\sum_{i=1}^n a_i x_i = 0$ with $a_i \in \mathbb{R} \subset \mathbb{C}$.
    Since $x_1, \ldots, x_n$ are linearly independent over $\mathbb{C}$, all $a_i = 0$.
    Since $x_1, \ldots, x_n$ generate $\mathbb{C}^n$ over $\mathbb{C}$ and each $x_i \in \mathbb{R}^n$,
    for any $v \in \mathbb{R}^n \subset \mathbb{C}^n$ we have $v = \sum_{i=1}^n c_i x_i$ with $c_i \in \mathbb{C}$.
    Taking real parts gives $v = \sum_{i=1}^n a_i x_i$ with $a_i \in \mathbb{R}$, so $x_1, \ldots, x_n$ generate $\mathbb{R}^n$.
    Thus $x_1, \ldots, x_n$ is a basis of $\mathbb{R}^n$.
\end{proof}

\begin{tcolorbox}[title=Problem 23, breakable]
    Let $V$ and $W$ be finite-dimensional vector spaces,
    $V \times W$ the product vector space (1.3.11). Prove:
    \[dim(V \times W) = dim V + dim W.\]
    {Hint: If $x_1, \ldots, x_n$ is a basis of $V$ and 
    $y_1, \ldots, y_m$ is a basis of $W$, show that the vectors 
    \[(x_1, 0), \ldots, (x_n, 0), (0, y_1), \ldots, (0, y_m),\]
    are a basis of $V \times W$, cf. 3.1, Exercise 6.}
\end{tcolorbox}

\begin{proof}
    Suppose $x_1, \ldots, x_n$ is a basis of $V$ and 
    $y_1, \ldots, y_m$ is a basis of $W$.
    Consider the set
    \[(x_1, 0), \ldots, (x_n, 0), (0, y_1), \ldots, (0, y_m).\]
    Let $(v,w) \in V \times W$. Then
    \[v = \sum a_i x_i \text{ and } w = \sum b_j y_j.\]
    Thus
    \[(v,w) = \sum a_i (x_i, 0) + \sum b_j (0, y_j),\]
    so the set is generating.
    Now suppose
    \[\sum a_i (x_i, 0) + \sum b_j (0, y_j) = (0,0).\]
    Then
    \[(\sum a_i x_i, \sum b_j y_j) = (0,0).\]
    Thus $\sum a_i x_i = 0$ and $\sum b_j y_j = 0$.
    Since $x_1, \ldots, x_n$ and $y_1, \ldots, y_m$ are bases,
    all $a_i, b_j = 0$.
    Thus the set is linearly independent.
    Therefore it is a basis of $V \times W$, so
    \[dim(V \times W) = dim V + dim W.\]
\end{proof}


\begin{tcolorbox}[title=Problem 24, breakable]
    If $V$ is a finite-dimensional vector space and $M$ 
    is a linear subspace of $V$, then there exists a linear subspace $N$ of $V$ 
    such that $V = M \oplus N$ in the sense of 1.6 Exercise 12. {Hint: Corollary 3.5.16.}
\end{tcolorbox}

\begin{proof}
    Let $x_1, \ldots, x_r$ be a basis for $M$.
    By Corollary 3.5.16, $x_1, \ldots, x_r$ can be expanded 
    to be a basis of $V$. Suppose the expanded basis is 
    \[x_1, \ldots, x_r, x_{r+1}, \ldots, x_n.\]
    Let $N = \operatorname{span}(x_{r+1}, \ldots, x_n)$.
    Now, let $v \in V$.
    Then for chosen $c_1, \ldots, c_n$ we have 
    \[v = \sum_{i=1}^{n} c_i x_i = \sum_{i=1}^{r} c_i x_i + \sum_{i=r+1}^{n} c_i x_i \in M + N.\]
    Now suppose $v \in M \cap N$.
    Then 
    \[v = \sum_{i=1}^{r} c_i x_i \text{ and } v = \sum_{i=r+1}^{n} c_i x_i \implies \sum_{i=1}^{r} c_i x_i - \sum_{i=r+1}^{n} c_i x_i = 0.\]
    Since $x_1, \ldots, x_n$ are linearly independent we must have $c_1 = \cdots = c_n = 0$.
    Thus $v = 0$, so $M \cap N = \{0\}$.
    It follows that $V = M \oplus N$.
\end{proof}


\begin{tcolorbox}[title=Problem 25, breakable]
    As in Theorem 3.4.8, let $V$ be a vector space,
    $x_1, \ldots, x_r$, $x_{r + 1}, \ldots, x_n \in V$,
    $M = [\{x_{r + 1}, \ldots, x_n\}]$ and $Q : V \longrightarrow V / M$
    the quotient mapping. The following conditions are equivalent:
    \begin{enumerate}
        \item \textcircled{a} $x_1, \ldots, x_n$ is a basis of $V$;
        \item \textcircled{b} $Q x_1, \ldots, Q x_n$ is a basis of $V / M$ and $x_{r + 1}, \ldots, x_n$ is a basis of $M$.
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    \underbar{\textbf(\textcircled{a}$\longrightarrow$\textcircled{b})}
    Suppose $x_1, \ldots, x_n$ is a basis of $V$.
    By 3.4.8, $Qx_1, \ldots, Qx_r$ generate $V/M$.
    For contradiction suppose for some $i$ with $1 \le i \le r$
    we have $Qx_i = \sum_{j \ne i} c_j Qx_j$.
    Then
    \[Qx_i - \sum_{j \ne i} c_j Qx_j = 0 \implies Q\!\left(x_i - \sum_{j \ne i} c_j x_j\right) = 0 \implies x_i - \sum_{j \ne i} c_j x_j \in \ker Q = M.\]
    So $x_i - \sum_{j \ne i} c_j x_j = \sum_{k=r+1}^{n} d_k x_k$ for some scalars $d_k$,
    which contradicts that $x_1, \ldots, x_n$ are linearly independent.
    It follows that $Qx_1, \ldots, Qx_r$ is a basis of $V/M$.
    Since $x_1, \ldots, x_n$ is a basis of $V$ and $M = [\{x_{r+1}, \ldots, x_n\}]$,
    it is clear that $x_{r+1}, \ldots, x_n$ is a basis of $M$.

    \underbar{\textbf(\textcircled{a}$\longleftarrow$\textcircled{b})}
    Suppose $Qx_1, \ldots, Qx_r$ is a basis of $V/M$ and $x_{r+1}, \ldots, x_n$ is a basis of $M$.
    Let $v \in V$. Since $Qx_1, \ldots, Qx_r$ generate $V/M$, we have
    \[Qv = \sum_{i=1}^{r} c_i Qx_i = Q\!\left(\sum_{i=1}^{r} c_i x_i\right),\]
    so $v - \sum_{i=1}^{r} c_i x_i \in \ker Q = M$.
    Since $x_{r+1}, \ldots, x_n$ generate $M$, we have $v - \sum_{i=1}^{r} c_i x_i = \sum_{i=r+1}^{n} c_i x_i$,
    thus $v = \sum_{i=1}^{n} c_i x_i$ and $x_1, \ldots, x_n$ span $V$.
    Now suppose $\sum_{i=1}^{n} c_i x_i = 0$. Then
    \[Q\!\left(\sum_{i=1}^{n} c_i x_i\right) = 0 \implies \sum_{i=1}^{r} c_i Qx_i = 0.\]
    Since $Qx_1, \ldots, Qx_r$ are linearly independent we must have $c_1 = \cdots = c_r = 0$.
    Thus $\sum_{i=r+1}^{n} c_i x_i = 0$, and since $x_{r+1}, \ldots, x_n$ are linearly independent
    we must have $c_{r+1} = \cdots = c_n = 0$.
    Thus $x_1, \ldots, x_n$ are linearly independent and are thus a basis of $V$.
\end{proof}

\begin{tcolorbox}[title=Problem 26, breakable]
    Let $V$ be a finite-dimensional vector space,
    $M$ and $N$ linear subspace of $V$ such that 
    $dim M + dim N = dim V$. Prove: $M + N = V \iff M \cap N = \{0\}$.
    {Hint: Corollary 3.5.14, (ii) and Theorem 3.5.13 (2).}
\end{tcolorbox}

\begin{proof}
    Note that $M + N$ is a subspace of $V$ with $\dim(M + N) \le \dim V$.

    \underline{\textbf{($\longrightarrow$)}}
    Suppose $M + N = V$. Let $x_1, \ldots, x_r$ be a basis of $M \cap N$.
    By 3.5.16, extend to a basis $x_1, \ldots, x_r, m_1, \ldots, m_s$ of $M$
    and a basis $x_1, \ldots, x_r, n_1, \ldots, n_t$ of $N$.
    Then $x_1, \ldots, x_r, m_1, \ldots, m_s, n_1, \ldots, n_t$ generate $M + N = V$,
    so $r + s + t \ge \dim V = \dim M + \dim N = (r + s) + (r + t)$.
    Thus $r \le 0$ thus $r = 0$ and $M \cap N = \{0\}$.

    \underline{\textbf{($\longleftarrow$)}}
    Suppose $M \cap N = \{0\}$. Let $m_1, \ldots, m_s$ and $n_1, \ldots, n_t$
    be bases of $M$ and $N$ respectively.
    Then $s + t = \dim V$.
    Then $m_1, \ldots, m_s, n_1, \ldots, n_t$ generate $M + N$.
    These vectors are linearly independent so suppose $\sum a_i m_i + \sum b_j n_j = 0$
        then $\sum a_i m_i = -\sum b_j n_j \in M \cap N = \{0\}$,
        so all $a_i = b_i = 0$.
    Thus $m_1, \ldots, m_s, n_1, \ldots, n_t$ is a basis of $M + N$ of size $\dim V$,
        so by 3.5.14 (ii) $M + N = V$.
\end{proof}

\begin{tcolorbox}[title=Problem 27, breakable]

\end{proof}

\begin{tcolorbox}[title=Problem 1, breakable]
    Does there exist a linear mapping $T : \mathbb{R}^7 \longrightarrow \mathbb{R}^3$
    whose kernel is 3-dimensional?
\end{tcolorbox}

\begin{proof}
    Here, $R = 3$ and $D = 7$ so $N = 4$ so no.
\end{proof}


\begin{tcolorbox}[title=Problem 2, breakable]
    Let $V$ be a finite-dimensional vector space, $M$ and $N$ linear 
    subspaces of $V$. Prove that $dim M + dim N = dim(M + N) + dim(M \cap N)$.
\end{tcolorbox}

\begin{proof}
    Consider the mapping $T : M \longrightarrow (M + N) / N$ defined 
    \[Tx = x + N \text{ for all } x \in M.\]
    By 2.7 Problem 1 we know $ran T = (M + N) / N$ and $\ker T = M \cap N$.
    Clearly $dom T = M$. Applying the rank-nullity theorem 
    \[\dim((M + N) / N) + \dim(M \cap N) = \dim(M).\]
    Then since $\dim((M + N) / N) = \dim(M + N) - \dim(N)$ we have 
    \[\dim(M + N) - \dim(N) + \dim(M \cap N) = \dim(M),\]
    so 
    \[\dim(M + N) + \dim(M \cap N) = \dim(M) + \dim(N).\]
\end{proof}

\begin{tcolorbox}[title=Problem 3, breakable]
    Let $V$ be a vector space over $F$, $f : V \longrightarrow F$ a linear form on $V$,
    $N$ the kernel of $f$. Prove that if $f$ is not identically zero, then 
    $V / N \cong F$.
\end{tcolorbox}

\begin{proof}
    Suppose $f$ is not identically zero, so there exists $z \in V$ with $f(z) \ne 0$.
    Let $c = f(z) \in F$. Then $f(c^{-1} z) = c^{-1} f(z) = 1$, so $f$ is surjective.
    By the First Isomorphism Theorem 
    \[V / \ker f = V / N \cong ran f = F.\]
\end{proof}

\begin{tcolorbox}[title=Problem 4, breakable]
    Let $V$ be a finite-dimensional vector space, $M$ and $N$ 
    linear subspaces of $V$ such that the dimensions of 
    \[(M + N)/N, (M + N)/M, M \cap N,\]
    are $2, 3, 4$, respectively. Find the dimensions of $M, N$ and $M + N$.
\end{tcolorbox}

\begin{proof}
    From the given dimensions we have the relations
    \[\dim(M + N) - \dim(N) = 2, \quad \dim(M + N) - \dim(M) = 3, \quad \dim(M \cap N) = 4.\]
    By Problem 2,
    \[\dim(M + N) + \dim(M \cap N) = \dim(M) + \dim(N),\]
    and substituting $\dim(M + N) = \dim(N) + 2$ and $\dim(M \cap N) = 4$ gives
    \[(\dim(N) + 2) + 4 = \dim(M) + \dim(N),\]
    so $\dim(M) = 6$. Substituting back,
    \[\dim(M + N) = \dim(M) + 3 = 9, \qquad \dim(N) = \dim(M + N) - 2 = 7. \qquad \qedhere\]
\end{proof}

\begin{tcolorbox}[title=Problem 6, breakable]
    The converse of Theorem 3.6.4 is true: If a product space $V \times W$
    is finite-dimensional, then so are the factor spaces $V$ and $W$.
\end{tcolorbox}

\begin{proof}
    Suppose $V \times W$ is finite-dimensional. Since $V \times \{0\}$ and 
    $\{0\} \times W$ are subspaces of $V \times W$, they are also finite-dimensional.
    Since $V \cong V \times \{0\}$ and $W \cong \{0\} \times W$, it follows that 
    $V$ and $W$ are finite-dimensional.
\end{proof}

\begin{tcolorbox}[title=Problem 7, breakable]
    Let $V$ be an $n$-dimensional vector space and let $f$ 
    be a nonzero linear form on $V$, that is, a nonzero element of the 
    dual space $V'$ of $V$. Prove that the kernel of $f$ is $(n - 1)$-dimensional.
    Conversely, every $(n - 1)$-dimensional linear subspace of $V$ is the kernel of a linear form.
\end{tcolorbox}

\begin{proof}
    Since $f$ is a nonzero linear form, $\ran f = F$ and so $\dim \ran f = 1$.
    By the First Isomorphism Theorem,
    \[V / \ker f \cong \ran f \implies \dim V - \dim \ker f = \dim \ran f = 1 \implies \dim \ker f = n - 1.\]
    
    Conversely, let $N \subseteq V$ be an $(n-1)$-dimensional subspace. 
    Let $\{e_1, \ldots, e_{n-1}\}$ be a basis for $N$ and extend it to a basis 
    $\{e_1, \ldots, e_{n-1}, e_n\}$ of $V$. Define $f : V \to F$ by 
    \[f\left(\sum_{i=1}^n a_i e_i\right) = a_n.\]
    Then $f$ is a nonzero linear form with $\ker f = N$.
\end{proof}


\begin{tcolorbox}[title=Problem 8, breakable]
    If $V$ is an $n$-dimensional vector space and if $f, g$ are linear forms on $V$
    that are not proportional (neither is a multiple of the other), prove that 
    $(\ker f) \cap (\ker g)$ is $(n - 2)$-dimensional.
\end{tcolorbox}

\begin{proof}
    Suppose $V$ is an $n$-dimensional vector space and $f, g$ are linear forms on $V$
    that are not proportional.
    By Problem 7, we know that $\dim \ker f = \dim \ker g = n - 1$.
    By Problem 16, $g$ is proportional to $f$ if and only if $g(x) = 0$ for all $x \in \ker f$.
    Thus there exists $x \in \ker f$ such that $g(x) \ne 0$, so the restriction 
    $g|_{\ker f} : \ker f \to F$ is a nonzero linear form on the $(n-1)$-dimensional 
    space $\ker f$. By Problem 7,
    \[\dim \ker(g|_{\ker f}) = (n - 1) - 1 = n - 2.\]
    But $\ker(g|_{\ker f}) = \ker f \cap \ker g$, so $\dim(\ker f \cap \ker g) = n - 2$.
\end{proof}

\begin{tcolorbox}[title=Problem 9, breakable]
    Let $V$ be a vector space and suppose $f_1, \ldots, f_n$ are linear forms 
    on $V$ such that 
    \[(\ker f_1) \cap \ldots \cap (\ker f_n) = \{0\}.\]
    Prove: $V$ is a finite-dimensional and $\dim V \le n$.
\end{tcolorbox}

\begin{proof}
    Define $T : V \to F^n$ by $Tx = (f_1(x), \ldots, f_n(x))$. Then $T$ is linear and 
    $\ker T = (\ker f_1) \cap \ldots \cap (\ker f_n) = \{0\}$, so $T$ is injective.
    Thus $V \cong \ran T \subseteq F^n$, so $V$ is finite-dimensional and $\dim V \le n$.
\end{proof}


\begin{tcolorbox}[title=Problem 14, breakable]
    Let $T : V \longrightarrow W$ and $S : W \longrightarrow U$ be linear mappings,
    with $V$ finite-dimensional.
    \begin{enumerate}
        \item If $S$ is injective, then $\ker ST = \ker T$ and $\rho (ST) = \rho(T)$.
        \item If $T$ is surjective, then $\ran ST =\ran S$ and $v(ST) - v(S) = \dim V - \dim W$.
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    Suppose $S$ is injective.
    Let $v \in \ker ST$, then $(ST)(v) = S(T(v)) = 0$.
    Since $S$ is injective, $T(v) = 0$, so $v \in \ker T$.
    Conversely, if $v \in \ker T$ then $(ST)(v) = S(T(v)) = S(0) = 0$, so $v \in \ker ST$.
    Thus $\ker ST = \ker T$, and by rank-nullity $\rho(ST) = \rho(T)$.

    Suppose $T$ is surjective. For any $u \in \ran S$, there exists $w \in W$ such that 
    $u = S(w)$. Since $T$ is surjective, there exists $v \in V$ such that $T(v) = w$, 
    so $u = S(T(v)) = (ST)(v) \in \ran ST$. Thus $\ran ST = \ran S$.
    By rank-nullity applied to $ST$ and $S$,
    \[v(ST) = \dim V - \rho(ST) = \dim V - \rho(S) = \dim V - (\dim W - v(S)),\]
    so $v(ST) - v(S) = \dim V - \dim W$.
\end{proof}


\begin{tcolorbox}[title=Problem 15, breakable]
    Let $V$ be a finite-dimensional vector space, $T \in \mathcal{L}(V)$.
    The following are equivalent:
    \begin{enumerate}
        \item \textcircled{a} $V = \ker T +\ran T$;
        \item \textcircled{b} $\ker T \cap \ran T = \{0\}$;
        \item \textcircled{c} $V = \ker T \oplus\ran T$.
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    \underbar{\textbf{(\textcircled{a}$\longrightarrow$\textcircled{b})}}
    Suppose $V = \ker T + \ran T$.
    \[\dim V = \dim \ker T + \dim \ran T,\]
    and since $V = \ker T + \ran T$ by the dimension formula from Problem 2
    \[\dim V = \dim \ker T + \dim \ran T - \dim(\ker T \cap \ran T),\]
    so $\dim(\ker T \cap \ran T) = 0$ thus $\ker T \cap \ran T = \{0\}$.

    \underbar{\textbf{(\textcircled{b}$\longrightarrow$\textcircled{c})}}
    Suppose $\ker T \cap \ran T = \{0\}$. By rank-nullity
    \[\dim(\ker T + \ran T) = \dim \ker T + \dim \ran T - \dim(\ker T \cap \ran T) = \dim V - 0 = \dim V,\]
    so $\ker T + \ran T = V$. Since $\ker T \cap \ran T = \{0\}$ we have 
    $V = \ker T \oplus \ran T$.

    \underbar{\textbf{(\textcircled{c}$\longrightarrow$\textcircled{a})}}
    This is immediate since $V = \ker T \oplus \ran T$ implies $V = \ker T + \ran T$
    by definition of the direct sum.
\end{proof}

\begin{tcolorbox}[title=Problem 3, breakable]
    Let $V$ be a vector space and let $S, T \in \mathcal{L}(V)$. Prove:
    \begin{enumerate}
        \item[(i)] $\rho(S + T) \le \rho(S) + \rho(T)$. (Hint: \S 3.6, Exercise 2.)
        \item[(ii)] $\rho(ST) \le \rho(S)$ and $\rho(ST) \le \rho(T)$.
        \item[(iii)] $\nu(S + T) \ge \nu(S) + \nu(T) - \dim V$.
    \end{enumerate}
\end{tcolorbox}

\begin{proof}
    We have 
    \[\rho(S) + \rho(T) = \dim(\ran S + \ran T) + \dim(\ran S \cap \ran T).\]
    So $\rho(S + T) = \rho(S) + \rho(T) - \dim(\ran S \cap \ran T)$.
    Then 
    \[\rho(S) + \rho(T) - \rho(S + T) = \rho(S) + \rho(T) - [\rho(S) + \rho(T) - \dim(\ran S \cap \ran T)] = \dim(\ran S \cap \ran T) \ge 0.\]
    Thus $\rho(S + T) \le \rho(S) + \rho(T)$.
\end{proof}

\begin{proof}
    Since $(ST)(x) = S(T(x)) \in \ran S$ for any $x \in V$, we have 
    $\ran(ST) \subseteq \ran S$, so $\rho(ST) \le \rho(S)$.
    Since $\ran(ST) = S(\ran T)$ and $S|_{\ran T} : \ran T \to V$ is linear,
    we have $\rho(ST) = \dim S(\ran T) \le \dim \ran T = \rho(T)$.
\end{proof}

\begin{proof}
    By rank-nullity $\nu(S) = \dim V - \rho(S)$ and $\nu(T) = \dim V - \rho(T)$. Then
    \[\nu(S + T) = \dim V - \rho(S + T) \ge \dim V - \rho(S) - \rho(T) = \nu(S) + \nu(T) - \dim V.\]
\end{proof}

\begin{tcolorbox}[title=Problem 6, breakable]
    Let $\mathcal{D}$ be the vector space of all real polynomial functions (Example 1.3.5).
    As in Examples 2.1.4 and 2.1.5, let $S, T \in \mathcal{L}(V)$ be the linear mappings 
    such that $Tp = p'$ and $Sp$ is the prederivative of $p$ with constant term $0$.

\begin{proof}
    $ST$ is not bijective. Consider any nonzero constant polynomial $f = c_0 \ne 0$.
    Then $T(f) = f' = 0$, so $ST(f) = S(0) = 0$.

\end{proof}

\begin{tcolorbox}[title=Problem 7, breakable]
    Let $V$ be any vector space (not necessarily finite-dimensional) and suppose 
    $R, S, T \in \mathcal{L}(V)$ are such that $ST = I$ and $TR = I$. Prove that 
    $T$ is bijective and $R = S = T^{-1}$. (Hint: Look at $S(TR)$.)
\end{tcolorbox}

\begin{proof}
    We have 
    \[TR = I \implies S(TR) = S \implies (ST)R = S \implies IR = S \implies R = S.\]
    Since $R = S$ and $TR = I$, we have $TS = I$. Combined with $ST = I$, 
    we have $ST = TS = I$, so $T$ is bijective and $R = S = T^{-1}$.
\end{proof}

\begin{tcolorbox}[title=Problem 9, breakable]
    Let $V$ be a finite-dimensional vector space and let $S, T \in \mathcal{L}(V)$. Prove:
    \begin{enumerate}
        \item[(i)] If there exists a nonzero vector $x$ such that $STx = \theta$, then 
        there exists a nonzero vector $y$ such that $TSy = \theta$.
        \item[(ii)] $ST$ is bijective $\Leftrightarrow$ $S$ and $T$ are bijective $\Leftrightarrow$ $TS$ is bijective.
    \end{enumerate}
    (Hint: Exercise 3.)
\end{tcolorbox}

\begin{proof}
    Suppose there exists a nonzero vector $x$ such that $STx = \theta$.
    There are two cases.
    If $Tx = \theta$, then $x \ne \theta$ implies $\nu(T) \ge 1$, and by Exercise 3, 
    $\nu(TS) \ge \nu(T) \ge 1$.
    If $Tx \ne \theta$, then since $S(Tx) = \theta$ with $Tx \ne \theta$, 
    we have $\nu(S) \ge 1$, and by Exercise 3, $\nu(TS) \ge \nu(S) \ge 1$.
\end{proof}

\begin{proof}
    Suppose $ST$ is bijective.
    If $T$ is not injective, there exist $x_1 \ne x_2$ with $T(x_1) = T(x_2)$,
    so $ST(x_1) = ST(x_2)$, contradicting injectivity of $ST$.
    If $S$ is not surjective, there exists $y \in V$ with $y \notin \ran S$,
    so $y \notin \ran(ST)$, contradicting surjectivity of $ST$.
    Thus $T$ is injective and $S$ is surjective. Since $V$ is finite-dimensional,
    both $S$ and $T$ are bijective.

    Conversely, suppose $S$ and $T$ are bijective. Then $(ST)^{-1} = T^{-1}S^{-1}$,
    so $ST$ is bijective.

    Thus $TS$ is bijective $\Leftrightarrow$ $T$ and $S$ are bijective $\Leftrightarrow$ $ST$ is bijective.
\end{proof}

\begin{tcolorbox}[title=Exercise 6, breakable]
    Here is a sketch of an alternate proof of Corollary 3.8.2, 
    based on constructing an explicit basis for $\mathcal{L}(V, W)$. Let $x_1, \ldots, x_n$ 
    be a basis of $V$ and $y_1, \ldots, y_m$ a basis of $W$. For each $i \in \{1, \ldots, m\}$ 
    and each $j \in \{1, \ldots, n\}$, there exists (by Theorem 3.8.1) a linear mapping 
    $E_{ij} : V \to W$ such that
    \[E_{ij} x_j = y_i \quad \text{and} \quad E_{ij} x_k = \theta \quad \text{for } k \ne j.\]
    Show that the $mn$ linear mappings constructed in this way are a basis of 
    $\mathcal{L}(V, W)$. (Verify that the $E_{ij}$ are linearly independent and that every 
    $T \in \mathcal{L}(V, W)$ is a linear combination of them.)
\end{tcolorbox}

\begin{proof}
    Let $x_1, \ldots, x_n$ 
    be a basis of $V$ and $y_1, \ldots, y_m$ a basis of $W$. For each $i \in \{1, \ldots, m\}$ 
    and each $j \in \{1, \ldots, n\}$, there exists (by Theorem 3.8.1) a linear mapping 
    $E_{ij} : V \to W$ such that
    \[E_{ij} x_j = y_i \quad \text{and} \quad E_{ij} x_k = \theta \quad \text{for } k \ne j.\]
    We first show linear independence.
    Suppose, for contradiction, there exist scalars $c_{ij}$, not all zero, such that 
    \[\sum_{i,j} c_{ij} E_{ij} = \sum_{i=1}^m \sum_{j=1}^n c_{ij} E_{ij} = 0.\]
    Let $x_l$ be a basis element of $V$.
    Applying this transformation to $x_l$ gives
    \[\sum_{i=1}^m \sum_{j=1}^n c_{ij} E_{ij}(x_l) = \theta.\]
    Since $E_{ij}(x_l) = \theta$ for $j \ne l$, all terms with $j \ne l$ vanish, so we can simplify to
    \[\sum_{i=1}^m c_{il} E_{il}(x_l) = \sum_{i=1}^m c_{il}\, y_i = \theta.\]
    Since $y_1, \ldots, y_m$ is a basis of $W$, they are linearly independent, so $c_{il} = 0$ for all $i$.
    Since $l$ was arbitrary, $c_{ij} = 0$ for all $i, j$, a contradiction.
    
    We now show generation.
    Let $T \in \mathcal{L}(V, W)$.
    Now, $T$ is completely determined by where it sends $x_1, \ldots, x_n$.
    For each basis element $x_j$, since $Tx_j \in W$, we can write
    \[T x_j = \sum_{i=1}^m a_{ij}\, y_i\]
    for some scalars $a_{ij}$. Consider the map $\sum_{i,j} a_{ij} E_{ij}$.
    Applying it to any basis element $x_l$ gives
    \[\sum_{i=1}^m \sum_{j=1}^n a_{ij} E_{ij}(x_l) = \sum_{i=1}^m a_{il}\, y_i = T x_l.\]
    Since $T$ and $\sum_{i,j} a_{ij} E_{ij}$ agree on every basis element of $V$ they are equal.
\end{proof}

\begin{tcolorbox}[title=Exercise 8, breakable]
    It is an exercise in concentration to calculate the transpose of a linear 
    form $f : V \to F$.
    \begin{enumerate}
        \item[(i)] For each $a \in F$, the formula $a'(c) = ac$ defines a linear form $a'$ on 
        $F$, and $a \mapsto a'$ is an isomorphism $F \to F'$.
        \item[(ii)] $a'f$ (the composite of $a'$ and $f$) is equal to $af$ (the scalar multiple 
        of $f$ by the scalar $a$).
        \item[(iii)] $f'a' = af$ for all $a \in F$.
    \end{enumerate}
\end{tcolorbox}

\begin{tcolorbox}[title=Exercise 14, breakable]

    \[{}^\circ N = \{x \in V : f(x) = 0 \text{ for all } f \in N\},\]
    that is, ${}^\circ N$ is the intersection of the kernels of all the $f \in N$. \{This is a 
    sort of `backward annihilator', analogous to the `forward annihilator' $M^\circ$ 
    of Definition 3.9.10; the effect on subspaces is that $M \mapsto M^\circ$ moves us 
    forward in the list $V, V', V'', V''', \ldots$ and $N \mapsto {}^\circ N$ moves us backward.\}
    Prove that ${}^\circ(M^\circ) = M$ for every linear subspace $M$ of $V$.
    \{Hint: The inclusion $M \subset {}^\circ(M^\circ)$ is trivial; for the reverse inclusion, see 
    \S 3.8, Exercise 13. Alternate proof: With $S : V \to V''$ as in 3.9.15, let 
    $(M^\circ)^\circ$ be the annihilator of $M^\circ$ in $V''$; show that $S(M) \subset (M^\circ)^\circ$ then 
    argue that $S(M) = (M^\circ)^\circ$ on the grounds of dimensionality (3.9.11), and 
    finally observe that ${}^\circ(M^\circ) = S^{-1}((M^\circ)^\circ)$.\}
